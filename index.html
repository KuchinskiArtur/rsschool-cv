            function get_pending_uninstalls(customer_id, table_name, callback)
            {
                var pendingUninstalls = {};
                var nextBatch = function(lek) {
                    var params = {  TableName: table_name,
                                    ExclusiveStartKey: lek,
                                    KeyConditionExpression: "#CI = :ci",
                                    ExpressionAttributeNames:{
                                        "#CI": "customer_id"
                                    },
                                    ExpressionAttributeValues: {
                                        ":ci": {"S": customer_id},
                                    }};
                    a1_dynamodb.query(params, function(err, data) {
                        if (err)
                        {
                            console.log(err, err.stack);
                            err.message = "ERROR: " + err.message;
                            callback(err);
                            return;
                        }
                        for (var item in data.Items)
                        {
                            if ( data.Items[item].hasOwnProperty("id") )
                            {
                                pendingUninstalls[data.Items[item].id["S"]] = 1;
                            }
                        }
                        if (data.LastEvaluatedKey)
                        {
                            nextBatch(data.LastEvaluatedKey);
                        } else
                        {
                            callback(null, pendingUninstalls);
                        }
                    });
                };
                nextBatch();
            }
